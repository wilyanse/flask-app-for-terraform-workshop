#!/bin/bash
sudo yum update -y
sudo yum install -y docker
sudo systemctl start docker
sudo usermod -aG docker $USER
sudo docker pull wilyanse/aws-flask-app:latest
sudo docker rm -f aws-flask-app || true
sudo docker run -d --restart unless-stopped \
  -p 80:5000 \
  -e AWS_REGION=ap-southeast-2 \
  -e DYNAMODB_TABLE_NAME=products-table \
  -e S3_BUCKET_NAME=will-image-bucket \
  wilyanse/aws-flask-app:latest
